    <script>
        const voiceBtn = document.getElementById('voiceBtn');
        const waveform = document.getElementById('waveform');
        const transcription = document.getElementById('transcription');
        const transcriptText = document.getElementById('transcriptText');
        
        let isRecording = false;
        
        voiceBtn.addEventListener('click', () => {
            isRecording = !isRecording;
            
            if (isRecording) {
                waveform.classList.remove('hidden');
                transcription.classList.remove('hidden');
                voiceBtn.classList.add('scale-110');
                
                // Simulate transcription
                setTimeout(() => {
                    transcriptText.textContent = '"Create a modern restaurant website with a dark theme, reservation system, and menu gallery"';
                }, 1500);
                
                // Auto-stop after 5 seconds
                setTimeout(() => {
                    isRecording = false;
                    voiceBtn.classList.remove('scale-110');
                }, 5000);
            } else {
                voiceBtn.classList.remove('scale-110');
            }
        });
    </script>
