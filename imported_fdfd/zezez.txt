import React, { useState } from 'react';
import { Mic, Shield, Zap, Box, Menu, X, ChevronRight, Sparkles } from 'lucide-react';

// Main App with Router
export default function CloudNexusApp() {
  const [currentPhase, setCurrentPhase] = useState('home');
  const [menuOpen, setMenuOpen] = useState(false);

  const phases = [
    {
      id: 'voice',
      name: 'Voice AI Builder',
      icon: Mic,
      color: 'from-indigo-500 to-purple-600',
      description: 'Generate sites with your voice',
      phase: 'Phase 1'
    },
    {
      id: 'security',
      name: 'Security Scanner',
      icon: Shield,
      color: 'from-emerald-500 to-green-600',
      description: 'Zero-trust vulnerability detection',
      phase: 'Phase 2'
    },
    {
      id: 'cdn',
      name: 'Predictive CDN',
      icon: Zap,
      color: 'from-blue-500 to-cyan-600',
      description: 'ML-powered performance optimization',
      phase: 'Phase 3'
    },
    {
      id: 'metaverse',
      name: 'Metaverse Builder',
      icon: Box,
      color: 'from-purple-500 to-pink-600',
      description: '3D spatial web experiences',
      phase: 'Phase 4'
    }
  ];

  const renderPage = () => {
    switch(currentPhase) {
      case 'voice':
        return <VoiceBuilderPage />;
      case 'security':
        return <SecurityScannerPage />;
      case 'cdn':
        return <PredictiveCDNPage />;
      case 'metaverse':
        return <MetaversePage />;
      default:
        return <HomePage phases={phases} onNavigate={setCurrentPhase} />;
    }
  };

  return (
    <div className="min-h-screen bg-[#0a0e1a] text-white">
      {/* Navigation Header */}
      <header className="fixed top-0 left-0 right-0 z-50 border-b border-white/10 backdrop-blur-xl bg-[#0a0e1a]/80">
        <div className="max-w-7xl mx-auto px-4 h-16 flex items-center justify-between">
          <button 
            onClick={() => setCurrentPhase('home')}
            className="flex items-center gap-3 hover:opacity-80 transition-opacity"
          >
            <div className="w-10 h-10 rounded-xl bg-gradient-to-br from-blue-500 to-purple-600 flex items-center justify-center">
              <Sparkles className="w-5 h-5" />
            </div>
            <div className="hidden sm:block">
              <h1 className="text-sm font-bold">CLOUD NEXUS</h1>
              <p className="text-xs text-gray-400">Vision 2030</p>
            </div>
          </button>

          {/* Desktop Menu */}
          <nav className="hidden md:flex items-center gap-2">
            {phases.map(phase => {
              const Icon = phase.icon;
              return (
                <button
                  key={phase.id}
                  onClick={() => setCurrentPhase(phase.id)}
                  className={`px-4 py-2 rounded-lg text-sm font-medium transition-all ${
                    currentPhase === phase.id 
                      ? 'bg-white/10 text-white' 
                      : 'text-gray-400 hover:text-white hover:bg-white/5'
                  }`}
                >
                  <Icon className="w-4 h-4 inline mr-2" />
                  {phase.name}
                </button>
              );
            })}
          </nav>

          {/* Mobile Menu Toggle */}
          <button 
            onClick={() => setMenuOpen(!menuOpen)}
            className="md:hidden p-2 hover:bg-white/10 rounded-lg"
          >
            {menuOpen ? <X className="w-5 h-5" /> : <Menu className="w-5 h-5" />}
          </button>
        </div>

        {/* Mobile Menu */}
        {menuOpen && (
          <div className="md:hidden border-t border-white/10 bg-[#0a0e1a] p-4">
            {phases.map(phase => {
              const Icon = phase.icon;
              return (
                <button
                  key={phase.id}
                  onClick={() => {
                    setCurrentPhase(phase.id);
                    setMenuOpen(false);
                  }}
                  className="w-full flex items-center gap-3 p-3 rounded-lg hover:bg-white/5 mb-2"
                >
                  <Icon className="w-5 h-5 text-blue-400" />
                  <div className="flex-1 text-left">
                    <p className="text-sm font-medium">{phase.name}</p>
                    <p className="text-xs text-gray-400">{phase.description}</p>
                  </div>
                </button>
              );
            })}
          </div>
        )}
      </header>

      {/* Main Content */}
      <main className="pt-16">
        {renderPage()}
      </main>
    </div>
  );
}
